// == VOIDCODE EXPORT: KING’S DILEMMA — STANDALONE v3.0 ==
// Mode: Self-contained game engine (no OS imports)
// Output cadence: EACH CALL RETURNS A FULL TURN SEQUENCE
// Turn structure (fixed order):
//   1) strip                — EntropyStrip pulse (λ/φ/descriptor)
//   2) prior_turn_effect    — What just applied (if any; else null)
//   3) values               — Meters, dualities, year, tags, queue
//   4) text                 — Narrative block for display
//   5) decision_prompt      — The question posed this turn
//   6) choices              — Selectable options [{id, title}]
// There is NO “next” command. The host calls `turn(choice_id?)` and gets a full turn.

meta {
  id          = "kings_dilemma_standalone_v3_0"
  executable  = false
  framing     = "engine_and_schema"
  cadence     = "full_turn_per_call"
}

/* -----------------------------------------------------------
   0) ASSUMED PRIMITIVES (provided by host/runtime)
   -----------------------------------------------------------
   clamp(x,min,max), compare(a,op,b), intersects(set,arr), subset(arr,set),
   argmax(obj), join(arr,sep), lastN(arr,n), list(set), map(arr,f),
   find(arr,pred), weightedChoice(items, rng), min(a,b), max(a,b),
   stddev(array<number>), JSON.parse/JSON.stringify
   RNG: Random(seed).roll(lo:int, hi:int) -> int
   Time: NOW() -> int (epoch ms)
   Hex: HEX(int) -> "XXXXXX"
*/

/* -----------------------------------------------------------
   1) MINIMAL RUNTIME + STRIP (no OS)
   ----------------------------------------------------------- */
module Runtime {
  state.λ = 0.42  // baseline ORDERED
  state.φ = 0

  export readEntropy(): float { return state.λ }
  export readPhi(): int { return state.φ }
  export setEntropy(x: float): void {
    if (x < 0.0) x = 0.0
    if (x > 1.0) x = 1.0
    state.λ = x
  }
  export tick(): int { state.φ += 1; return state.φ }
}

module EntropyStrip {
  export descriptor(λ: float): string {
    if (λ < 0.40) return "ORDERED"
    if (λ < 0.70) return "VARIANT"
    if (λ < 0.90) return "FRACTURED"
    return "COLLAPSE"
  }
  export hash(λ: float, φ: int): string {
    let val = int(λ * 1000) + φ
    return HEX(val % 1048576) // 6 hex chars
  }
  export pulse(): string {
    let λ = Runtime.readEntropy()
    let φ = Runtime.readPhi()
    let Σ = hash(λ, φ)
    let Δ = descriptor(λ)
    return "// ⌖ KDM::[Σ=" + Σ + "|λ=" + λ.toFixed(2) + "|φ=" + φ + "|Δ=" + Δ + "]"
  }
}

/* -----------------------------------------------------------
   2) STATE + SCHEMA (strict; cards importable)
   ----------------------------------------------------------- */
module Schema {

  // 2.1 Meters & Dualities
  type MeterState {
    stability:int, faith:int, treasury:int, army:int, people:int, legitimacy:int, dread:int
  }
  type Duality {
    mercy_justice:int, war_peace:int, piety_reason:int, tradition_reform:int, secrecy_transparency:int
  }

  // 2.2 Effect (standardized)
  // All fields optional; absent fields = no-op
  type Effect {
    delta?: {stability?:int, faith?:int, treasury?:int, army?:int, people?:int, legitimacy?:int, dread?:int}
    shift?: {mercy_justice?:int, war_peace?:int, piety_reason?:int, tradition_reform?:int, secrecy_transparency?:int}
    add_tags?: [string]
    remove_tags?: [string]
    enqueue?: [{in:int, effect:Effect, note?:string}]
    note?: string
  }

  // 2.3 Option (standardized)
  type Option {
    id:string
    title:string
    effect:Effect
    risk?: {chance:int, on_success:Effect, on_fail:Effect, cue?:string}
    cost_check?: {treasury?:int, army?:int, legitimacy?:int}
    locks_if?: [string]
    unlocks_if?: [string]
  }

  // 2.4 Appears rules (standardized)
  type AppearsIf {
    any_tags?: [string], all_tags?: [string],
    duality?: {axis:"mercy_justice"|"war_peace"|"piety_reason"|"tradition_reform"|"secrecy_transparency", min?:int, max?:int},
    threshold?: {meter:"stability"|"faith"|"treasury"|"army"|"people"|"legitimacy"|"dread", op:">"|"<"|">="|"<="|"==", value:int},
    year_min?:int, year_max?:int
  }

  // 2.5 Card (standardized)
  type Card {
    id:string
    title:string
    scene:string           // narrative seed
    panel:{banner:string, mood:string}
    decision_prompt:string // explicit question to pose
    appears_if?: AppearsIf
    weight:int             // for weighted choice
    on_draw?: Effect
    options:[Option, ...]  // >= 2 options recommended
  }

  // 2.6 TurnOutput (fixed order)
  type TurnOutput {
    strip:string
    prior_turn_effect: {
      source?:string, timing?:("choice"|"delayed"|"on_draw"),
      deltas?:object, shifts?:object, tags_added?:[string], tags_removed?:[string],
      notes?:[string], risk?:("success"|"fail"|null)
    } | null
    values: {
      year:int, meters:MeterState, duality:Duality,
      tags:[string], queued_events:int
    }
    text: {
      banner:string, mood:string, scene:string, narration?:string
    }
    decision_prompt:string
    choices:[{id:string, title:string}]
  }

  // 2.7 Card Template (authoring guide; import with Deck.load)
  export CardTemplate = Card{
    id:"package_id__unique_snake",
    title:"CARD TITLE",
    scene:"Short, evocative description.",
    panel:{banner:"Banner Title", mood:"tone or ambiance"},
    decision_prompt:"What do you do?",
    appears_if:{ // all optional; remove block if always available
      any_tags:["tag_a"] , all_tags:["tag_b","tag_c"],
      duality:{axis:"war_peace", min:-100, max:100},
      threshold:{meter:"treasury", op:"<=", value:50},
      year_min:1, year_max:999
    },
    weight:5,
    on_draw:Effect{ note:"Optional one-shot text/effect when drawn." },
    options:[
      Option{ id:"opt_a", title:"Do thing A", effect:Effect{ delta:{people:+2} } },
      Option{ id:"opt_b", title:"Do thing B", effect:Effect{ delta:{treasury:+4} },
              risk:{chance:45, on_success:Effect{delta:{stability:+2}},
                            on_fail:Effect{delta:{stability:-3}}, cue:"Risk cue"} }
    ]
  }
}

/* -----------------------------------------------------------
   3) DECK IMPORT (packs can be authored separately)
   ----------------------------------------------------------- */
module Deck {
  state.cards = []  // array<Schema.Card>

  export clear(): void { state.cards = [] }

  export load(pack: list): {ok:bool, errors:[string]} {
    // Shallow schema checks (required fields + types where cheap)
    let errs = []
    for (let i=0;i<pack.length;i++) {
      let c = pack[i]
      if (!c.id || !c.title || !c.scene || !c.panel || !c.decision_prompt || !c.weight || !c.options) {
        errs.push("Card["+i+"]: missing required field(s)")
        continue
      }
      if (!c.panel.banner || !c.panel.mood) errs.push("Card["+i+"]: panel requires banner+mood")
      if (c.options.length < 2) errs.push("Card["+i+"]: needs >= 2 options")
    }
    if (errs.length == 0) { state.cards = pack; return {ok:true, errors:[]} }
    return {ok:false, errors:errs}
  }

  export all(): list { return state.cards }
}

/* -----------------------------------------------------------
   4) GAME ENGINE (no commands; single-call full turn output)
   ----------------------------------------------------------- */
module Game {
  /* 4.1 Config */
  const START_STATE = {stability:50, faith:50, treasury:50, army:50, people:50, legitimacy:50, dread:20}
  const LOSS = {revolt_people:10, bankrupt_treasury:0, anathema_faith:5, legitimacy_crisis:15}
  const WIN  = {golden_age_all:80, reign_years:30}
  const TURN_LIMIT = 40

  /* 4.2 Realm (internal) */
  state.year = 1
  state.meters = START_STATE
  state.duality = {mercy_justice:0, war_peace:0, piety_reason:0, tradition_reform:0, secrecy_transparency:0}
  state.tags = set()
  state.event_queue = []   // [{in, effect, ref, note}]
  state.rng = null
  state.seed = 0
  state.is_over = false
  state.ending = null
  state.last_effect_applied = null   // shaped into prior_turn_effect

  /* 4.3 Init */
  export init(seed?): void {
    state.seed = seed ?? (NOW() % 1_000_000)
    state.rng  = Random(state.seed)
    state.year = 1
    state.meters = JSON.parse(JSON.stringify(START_STATE))
    state.duality = {mercy_justice:0, war_peace:0, piety_reason:0, tradition_reform:0, secrecy_transparency:0}
    state.tags = set()
    state.event_queue = []
    state.is_over = false
    state.ending = null
    state.last_effect_applied = null
    Runtime.setEntropy(0.42)
  }

  /* 4.4 Helpers */
  fn clampInt(x:int, lo:int, hi:int): int { return clamp(x, lo, hi) }

  fn applyEffect(e, source, timing, riskFlag:null|"success"|"fail") {
    if (!e) return
    let deltas = {}
    let shifts = {}
    let addT = [], remT = [], notes = []
    if (e.delta) {
      for (let k in e.delta) {
        let nv = clampInt(state.meters[k] + e.delta[k], 0, 100)
        deltas[k] = (nv - state.meters[k])
        state.meters[k] = nv
      }
    }
    if (e.shift) {
      for (let k in e.shift) {
        let nv = clamp(state.duality[k] + e.shift[k], -100, 100)
        shifts[k] = (nv - state.duality[k])
        state.duality[k] = nv
      }
    }
    if (e.add_tags) { for (let t of e.add_tags) { state.tags.add(t); addT.push(t) } }
    if (e.remove_tags) { for (let t of e.remove_tags) { state.tags.delete(t); remT.push(t) } }
    if (e.enqueue) {
      for (let q of e.enqueue) state.event_queue.push({in:q.in, effect:q.effect, ref:source, note:q.note})
    }
    if (e.note) notes.push(e.note)

    // record prior turn effect (consumed by output)
    state.last_effect_applied = {
      source: source, timing: timing, deltas: deltas, shifts: shifts,
      tags_added: addT, tags_removed: remT, notes: notes, risk: riskFlag
    }
  }

  fn applyQueuedEffects() {
    let next = []
    for (let q of state.event_queue) {
      q.in -= 1
      if (q.in <= 0) {
        applyEffect(q.effect, q.ref, "delayed", null)
        if (q.note) { state.last_effect_applied.notes.push(q.note) }
      } else next.push(q)
    }
    state.event_queue = next
  }

  fn appears(c): bool {
    let a = c.appears_if
    if (!a) return true
    if (a.year_min && state.year < a.year_min) return false
    if (a.year_max && state.year > a.year_max) return false
    if (a.any_tags && !intersects(state.tags, a.any_tags)) return false
    if (a.all_tags && !subset(a.all_tags, state.tags)) return false
    if (a.threshold) {
      let m = state.meters[a.threshold.meter]
      if (!compare(m, a.threshold.op, a.threshold.value)) return false
    }
    if (a.duality) {
      let v = state.duality[a.duality.axis]
      if (a.duality.min && v < a.duality.min) return false
      if (a.duality.max && v > a.duality.max) return false
    }
    return true
  }

  fn drawCard() {
    let pool = []
    for (let c of Deck.all()) if (appears(c)) pool.push({c:c, w:c.weight})
    if (pool.length == 0) return null
    return weightedChoice(pool, state.rng).c
  }

  fn eligible(opt): bool {
    if (opt.unlocks_if && !subset(opt.unlocks_if, state.tags)) return false
    if (opt.locks_if && intersects(opt.locks_if, state.tags)) return false
    if (opt.cost_check) {
      for (let k in opt.cost_check) if (state.meters[k] < opt.cost_check[k]) return false
    }
    return true
  }

  fn mapComplexityToEntropy(): float {
    let base = 0.42
    let dsum = (abs(state.duality.mercy_justice) + abs(state.duality.war_peace) + abs(state.duality.piety_reason) +
                abs(state.duality.tradition_reform) + abs(state.duality.secrecy_transparency)) / 500.0
    let qfactor = min(state.event_queue.length / 6.0, 0.25)
    let metrics = [state.meters.stability, state.meters.faith, state.meters.treasury, state.meters.army, state.meters.people, state.meters.legitimacy, state.meters.dread]
    let meter_var = (stddev(metrics) / 100.0) * 0.3
    return clamp(base + dsum + qfactor + meter_var, 0.0, 0.99)
  }

  fn endCheck() {
    if (state.meters.people     <= LOSS.revolt_people && state.meters.army <= 30) return epilogue("revolt", "The streets rise; your banners burn.")
    if (state.meters.treasury   <= LOSS.bankrupt_treasury)                           return epilogue("bankrupt", "Coffers hollow; creditors claim the crown.")
    if (state.meters.faith      <= LOSS.anathema_faith)                              return epilogue("anathema", "Anathema proclaimed; no loyal blade remains.")
    if (state.meters.stability  <= 0)                                               return epilogue("collapse", "The realm fractures into warring fiefs.")
    if (state.meters.legitimacy <= LOSS.legitimacy_crisis)                           return epilogue("usurpation", "A subtler steel and quill take your throne.")
    // Wins
    let m = state.meters
    if (m.stability>=WIN.golden_age_all && m.faith>=WIN.golden_age_all && m.treasury>=WIN.golden_age_all &&
        m.army>=WIN.golden_age_all && m.people>=WIN.golden_age_all && m.legitimacy>=WIN.golden_age_all) {
      return epilogue("golden_age", "A rare concord: peace, plenty, renown.")
    }
    if (TURN_LIMIT > 0 && state.year > TURN_LIMIT) return epilogue("longevity", "Your long reign fades into legend.")
  }

  fn epilogue(key, text) {
    state.is_over = true
    let tone = []
    tone.push(state.duality.mercy_justice < 0 ? "Merciful" : "Severe")
    tone.push(state.duality.war_peace < 0 ? "Martial" : "Conciliatory")
    tone.push(state.duality.piety_reason < 0 ? "Pious" : "Scholarly")
    tone.push(state.duality.tradition_reform < 0 ? "Traditional" : "Reforming")
    tone.push(state.duality.secrecy_transparency < 0 ? "Secretive" : "Open")
    state.ending = {
      summary: key + ": " + text + " Your reign is remembered as " + join(tone,", ") + ".",
      banners: tone, tags: list(state.tags), years: state.year, last_decisions: [] // host may track if desired
    }
  }

  fn updateSeason() {
    if (state.year % 2 == 1) { state.tags.add("winter"); state.tags.delete("summer") }
    else { state.tags.add("summer"); state.tags.delete("winter") }
  }

  /* 4.5 PUBLIC: one-call-per-turn
     - On first call (choice_id == null/undefined): initializes (if requested by host prior) & returns first card.
     - On subsequent calls: applies chosen option from previous card, advances year, returns new turn output.
  */
  state._current_card = null

  export turn(choice_id?): Schema.TurnOutput {
    // PRIOR CHOICE RESOLUTION (if provided)
    state.last_effect_applied = null
    if (choice_id && state._current_card && !state.is_over) {
      let card = state._current_card
      let opt = find(card.options, o => o.id == choice_id)
      if (opt && eligible(opt)) {
        let riskFlag = null
        let eff = JSON.parse(JSON.stringify(opt.effect))
        if (opt.risk) {
          let roll = state.rng.roll(1,100)
          if (roll <= opt.risk.chance) {
            eff = mergeEffects(eff, opt.risk.on_success); riskFlag = "success"
          } else {
            eff = mergeEffects(eff, opt.risk.on_fail);    riskFlag = "fail"
          }
        }
        applyEffect(eff, card.id, "choice", riskFlag)
      }
      // Year advances after a valid resolution attempt (even if ineligible was passed; host should gate)
      state.year += 1
    }

    // APPLY DELAYED CONSEQUENCES & SEASON; UPDATE ENTROPY & PULSE
    applyQueuedEffects()
    updateSeason()
    Runtime.setEntropy( mapComplexityToEntropy() )
    Runtime.tick() // increment φ for this full-turn emission
    let strip_line = EntropyStrip.pulse()

    // ENDING CHECK (may have been reached via queues)
    endCheck()
    if (state.is_over) {
      // Return final "ending" turn (no choices)
      return {
        strip: strip_line,
        prior_turn_effect: state.last_effect_applied,
        values: {
          year: state.year,
          meters: state.meters,
          duality: state.duality,
          tags: list(state.tags),
          queued_events: state.event_queue.length
        },
        text: {
          banner: "Epilogue",
          mood: "tapestry stillness",
          scene: state.ending.summary,
          narration: "Years ruled: " + state.year
        },
        decision_prompt: "—",
        choices: []
      }
    }

    // DRAW NEXT CARD
    let card = drawCard()
    if (!card) {
      epilogue("stasis","The realm drifts into quiet obscurity.")
      return {
        strip: strip_line,
        prior_turn_effect: state.last_effect_applied,
        values: {
          year: state.year,
          meters: state.meters,
          duality: state.duality,
          tags: list(state.tags),
          queued_events: state.event_queue.length
        },
        text: { banner:"Stasis", mood:"quiet dusk", scene:"No counsel arrives; candles burn low.", narration: state.ending.summary },
        decision_prompt: "—",
        choices: []
      }
    }

    // on_draw effect
    if (card.on_draw) applyEffect(card.on_draw, card.id, "on_draw", null)

    // cache as active
    state._current_card = card

    // OUTPUT FULL TURN SEQUENCE
    return {
      strip: strip_line,
      prior_turn_effect: state.last_effect_applied,
      values: {
        year: state.year,
        meters: state.meters,
        duality: state.duality,
        tags: list(state.tags),
        queued_events: state.event_queue.length
      },
      text: {
        banner: card.panel.banner,
        mood:   card.panel.mood,
        scene:  card.scene,
        narration: null
      },
      decision_prompt: card.decision_prompt,
      choices: map(card.options, o => ({id:o.id, title:o.title}))
    }
  }
}

/* -----------------------------------------------------------
   6) BOOT SEQUENCE (host usage)
   ----------------------------------------------------------- */
module KD_Entry {
  // Load base pack; allow host to load custom packs before/after
  export boot(seed?): string {
    let r = Deck.load(BaseCards.pack)
    if (!r.ok) return "Deck load errors: " + JSON.stringify(r.errors)
    Game.init(seed)
    return "KDM ready."
  }

  // One call = one full turn output. Pass choice_id from last output’s choices[].id
  export turn(choice_id?): Schema.TurnOutput {
    return Game.turn(choice_id)
  }
}
